lower_tier_items = ["SPLIT", "RIT", "ATTN", "LOCK"]

# Dictionary container to safely hold on/off boolean variables in memory
state_variables = {}

def on_toggle_action(name):
    """Callback function triggered every time a state swaps."""
    current_state = state_variables[name].get()
    print(f"System State Update: Widget [{name}] is now {'ON' if current_state else 'OFF'}")

# Generate and pack the toggle elements
for item in lower_tier_items:
    # 1. Create a variable bound to memory to preserve individual status
    state_variables[item] = tk.BooleanVar(value=False)
    
    # 2. Declare widget configuration as a themed Checkbutton
    btn_toggle = ttk.Checkbutton(
        utility_bar, 
        text=item, 
        variable=state_variables[item],
        style="ToggleGhost.TToolbutton",
        command=lambda name=item: on_toggle_action(name) # Safe parameter pass
    )
    btn_toggle.pack(side="left", padx=4, expand=True, fill="x")
